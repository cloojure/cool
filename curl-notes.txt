curl -G, --get              # append form data to url using '?' separator
curl -H, --header           # can (repeatedly) add header information
curl -i, --include          # or -i; include http-header in output
curl -v, --verbose          # or -v
curl -X, --request <cmd>    # or -X;  <cmd> = GET (default), POST, PUT
